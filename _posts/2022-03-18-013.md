## 1. Boj 2606번 바이러스
```python
from collections import deque
import sys
n = int(sys.stdin.readline().rstrip())
m = int(sys.stdin.readline().rstrip())
dic = {}
for i in range(n):
    dic[i+1] = set()
for _ in range(m):
    a,b =map(int, sys.stdin.readline().rstrip().split())
    dic[a].add(b)
    dic[b].add(a)

def bfs(start, dic):
    d = deque([start])
    while d:
        for i in dic[d.popleft()]:
            if i not in visit:
                visit.append(i)
                d.append(i)
visit = []
bfs(1,dic)
print(len(visit)-1)
```
